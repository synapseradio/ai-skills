Share: ${d.data.value}%`); }) .on("pointermove", function (event, d) { VizHelpers.showTooltip(root, event, `${d.data.segment}
Share: ${d.data.value}%`); }) .on("pointerleave", function () { g.selectAll(".slice").attr("opacity", 1).attr("d", arc); VizHelpers.hideTooltip(root); }) .on("focus", function (event, d) { g.selectAll(".slice").attr("opacity", 0.3); d3.select(this).attr("opacity", 1).attr("d", arcHover); const rect = this.getBoundingClientRect(); tooltip .style("opacity", 1) .html(`${d.data.segment}
