// Compute degree for each node const degreeMap = {}; nodes.forEach((n) => { degreeMap[n.id] = 0; }); links.forEach((l) => { degreeMap[l.source]++; degreeMap[l.target]++; }); nodes.forEach((n) => { n.degree = degreeMap[n.id]; }); const t = VizHelpers.tokens; // --- DIMENSIONS --- const width = 880; const height = 520; const groups = [...new Set(nodes.map((d) => d.group))]; const colors = [t["oc-blue-7"], t["oc-orange-8"], t["oc-teal-7"]]; const colorScale = d3.scaleOrdinal().domain(groups).range(colors); const radiusScale = d3 .scaleSqrt() .domain([1, d3.max(nodes, (d) => d.degree)]) .range([10, 26]); // --- SVG --- const svg = d3 .select(root.querySelector("svg")) .attr("viewBox", `0 0 ${width} ${height}`) .attr("width", "100%"); // Zoom container. The zoom behavior binds its wheel/drag listeners to this // chart's own svg, so the gesture stays inside the card and never hijacks page // scroll (the .viz card also sets overscroll-behavior: contain). const zoomG = svg.append("g"); svg.call( d3 .zoom() .scaleExtent([0.3, 4]) .on("zoom", (event) => zoomG.attr("transform", event.transform)), ); // --- FORCE SIMULATION --- const simulation = d3 .forceSimulation(nodes) .force( "link", d3 .forceLink(links) .id((d) => d.id) .distance(90) .strength(0.7), ) .force("charge", d3.forceManyBody().strength(-280)) .force("center", d3.forceCenter(width / 2, height / 2)) .force( "collision", d3.forceCollide().radius((d) => radiusScale(d.degree) + 8), ); // --- LINKS --- const linkSel = zoomG .append("g") .attr("stroke", t["oc-gray-3"]) .attr("stroke-width", 1.5) .selectAll("line") .data(links) .join("line") .attr("aria-hidden", "true"); // --- NODES --- const nodeG = zoomG.append("g"); const nodeGroups = nodeG .selectAll(".node-group") .data(nodes) .join("g") .attr("class", "node-group"); // Draggable circle const nodeCircles = nodeGroups .append("circle") .attr("class", "node") .attr("r", (d) => radiusScale(d.degree)) .attr("fill", (d) => colorScale(d.group)) .attr("opacity", 0.85) .attr("stroke", "white") .attr("stroke-width", 2) .attr("tabindex", 0) .attr("role", "listitem") .attr("aria-label", (d) => `${d.id} (${d.group}): degree ${d.degree}`); // Node labels nodeGroups .append("text") .attr("text-anchor", "middle") .attr("dy", (d) => radiusScale(d.degree) + 13) .attr("font-size", "11px") .attr("fill", t["oc-gray-7"]) .attr("pointer-events", "none") .text((d) => d.id); // --- DRAG --- function drag(simulation) { return d3 .drag() .on("start", (event, d) => { if (!event.active) simulation.alphaTarget(0.3).restart(); d.fx = d.x; d.fy = d.y; }) .on("drag", (event, d) => { d.fx = event.x; d.fy = event.y; }) .on("end", (event, d) => { if (!event.active) simulation.alphaTarget(0); d.fx = null; d.fy = null; }); } nodeCircles.call(drag(simulation)); // Hover / focus interactions nodeCircles .on("pointerenter", function (event, d) { nodeCircles.attr("opacity", 0.2); d3.select(this) .attr("opacity", 1) .attr("r", radiusScale(d.degree) + 3); // highlight connected links linkSel .attr("stroke", (l) => l.source.id === d.id || l.target.id === d.id ? t["oc-blue-7"] : t["oc-gray-3"], ) .attr("stroke-width", (l) => l.source.id === d.id || l.target.id === d.id ? 2.5 : 1.5, ); VizHelpers.showTooltip( root, event, `${d.id}
Group: ${d.group}
Degree: ${d.degree}`, ); }) .on("pointermove", function (event, d) { VizHelpers.showTooltip( root, event, `${d.id}
Group: ${d.group}
Degree: ${d.degree}`, ); }) .on("pointerleave", function (event, d) { nodeCircles.attr("opacity", 0.85).attr("r", (n) => radiusScale(n.degree)); linkSel.attr("stroke", t["oc-gray-3"]).attr("stroke-width", 1.5); VizHelpers.hideTooltip(root); }) .on("focus", function (event, d) { nodeCircles.attr("opacity", 0.2); d3.select(this) .attr("opacity", 1) .attr("r", radiusScale(d.degree) + 3); linkSel .attr("stroke", (l) => l.source.id === d.id || l.target.id === d.id ? t["oc-blue-7"] : t["oc-gray-3"], ) .attr("stroke-width", (l) => l.source.id === d.id || l.target.id === d.id ? 2.5 : 1.5, ); const rect = this.getBoundingClientRect(); const tip = d3.select(root.querySelector(".tooltip")); tip .style("opacity", 1) .html(`${d.id}
Group: ${d.group}
Degree: ${d.degree}`) .style("left", `${rect.x + rect.width / 2}px`) .style("top", `${rect.y - 52}px`); }) .on("blur", function (event, d) { nodeCircles.attr("opacity", 0.85).attr("r", (n) => radiusScale(n.degree)); linkSel.attr("stroke", t["oc-gray-3"]).attr("stroke-width", 1.5); VizHelpers.hideTooltip(root); }); // Keyboard navigation VizHelpers.navigateMarks(root, "ArrowLeft", "ArrowRight"); // --- SIMULATION TICK --- simulation.on("tick", () => { linkSel .attr("x1", (d) => d.source.x) .attr("y1", (d) => d.source.y) .attr("x2", (d) => d.target.x) .attr("y2", (d) => d.target.y); nodeGroups.attr("transform", (d) => `translate(${d.x},${d.y})`); }); // --- LEGEND --- const legend = svg.append("g").attr("transform", `translate(16, 16)`); groups.forEach((group, i) => { const row = legend.append("g").attr("transform", `translate(0, ${i * 20})`); row.append("circle").attr("r", 6).attr("fill", colorScale(group)).attr("opacity", 0.85); row .append("text") .attr("x", 12) .attr("y", 4) .attr("font-size", "12px") .attr("fill", t["oc-gray-7"]) .text(group); }); // --- DATA TABLE --- // Node table, then edge table (renderDataTable appends each to .data-table). VizHelpers.renderDataTable( root, nodes.map((d) => [d.id, d.group, d.degree]), ["Node", "Group", "Degree"], ); VizHelpers.renderDataTable( root, links.map((l) => [ typeof l.source === "object" ? l.source.id : l.source, typeof l.target === "object" ? l.target.id : l.target, ]), ["Source", "Target"], );
