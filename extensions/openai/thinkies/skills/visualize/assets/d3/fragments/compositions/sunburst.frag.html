` (${d.parent.data.name})` : ""; return `${d.data.name}${parent}: ${d.value} people, ${pct}%`; }); paths .on("pointerenter", function (event, d) { g.selectAll(".arc-path").attr("opacity", 0.3); d3.select(this).attr("opacity", 1); const pct = ((d.value / total) * 100).toFixed(1); const parent = d.depth > 1 ? `
${d.parent.data.name}` : ""; tooltip .style("opacity", 1) .html(`${d.data.name}${parent}
${d.value} people (${pct}%)`) .style("left", `${event.pageX + 12}px`) .style("top", `${event.pageY - 28}px`); }) .on("pointermove", function (event) { tooltip.style("left", `${event.pageX + 12}px`).style("top", `${event.pageY - 28}px`); }) .on("pointerleave", function () { g.selectAll(".arc-path").attr("opacity", 1); VizHelpers.hideTooltip(root); }) .on("focus", function (event, d) { g.selectAll(".arc-path").attr("opacity", 0.3); d3.select(this).attr("opacity", 1); const pct = ((d.value / total) * 100).toFixed(1); const parent = d.depth > 1 ? `
${d.parent.data.name}` : ""; const rect = this.getBoundingClientRect(); tooltip .style("opacity", 1) .html(`${d.data.name}${parent}
${d.value} people (${pct}%)`) .style("left", `${rect.x + rect.width / 2}px`) .style("top", `${rect.y - 40}px`); }) .on("blur", function () { g.selectAll(".arc-path").attr("opacity", 1); VizHelpers.hideTooltip(root); }); // Keyboard navigation paths.on("keydown", function (event) { const nodes = paths.nodes(); const idx = nodes.indexOf(this); if (event.key === "ArrowRight" && idx < nodes.length - 1) { event.preventDefault(); nodes[idx + 1].focus(); } else if (event.key === "ArrowLeft" && idx > 0) { event.preventDefault(); nodes[idx - 1].focus(); } }); // --- ARC LABELS (depth=1 only, if arc is wide enough) --- g.selectAll(".arc-label") .data(hierarchyRoot.descendants().filter((d) => d.depth === 1)) .join("text") .attr("class", "arc-label") .attr("transform", (d) => { const [x, y] = arc.centroid(d); return `translate(${x},${y})`; }) .attr("text-anchor", "middle") .attr("fill", "white") .attr("font-weight", "600") .text((d) => (d.x1 - d.x0 > 0.3 ? d.data.name : "")); // Center label g.append("text") .attr("text-anchor", "middle") .attr("dy", "0.35em") .attr("font-size", "13px") .attr("fill", t["oc-gray-7"]) .text(`${total} total`); // --- DATA TABLE --- const tableRows = []; hierarchy.children.forEach((dept) => { dept.children.forEach((team, j) => { const pct = ((team.value / total) * 100).toFixed(1); tableRows.push([j === 0 ? dept.name : "", team.name, team.value, `${pct}%`]); }); }); VizHelpers.renderDataTable(root, tableRows, ["Department", "Team", "Headcount", "% of Total"]);
